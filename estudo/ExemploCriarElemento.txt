		<script type="text/javascript">		
		 
			function removerDiv(botaoClicado)
			{
				if(botaoClicado.value === "Remove" )
				{
					botaoClicado = "Replace";
					var content2 = document.getElementById("temp");  
					content2.parentNode.removeChild(content2);
				} 
				else
				{
					botaoClicado = "Remove";
					var criandoNovamenteDiv = document.createElement('div');  
					criandoNovamenteDiv.id = 'temp';
					document.body.appendChild(criandoNovamenteDiv);
				}
			} 
		</script>
