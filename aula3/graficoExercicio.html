var gutter = 2; var xPosition = x; var yPosition = y; var barHeight = 0; var count = 0; while(count < weatherData.length) { //construindo apartir 2ª linha do grafico barHeight = -1 * weatherData[count].temperature; xPosition = xPosition + gutter + barWidth; box(xPosition, yPosition, barWidth, barHeight); text(xPosition - 20, yPosition - 20 ,weatherData[count].date); if(count == 0) { //construindo 1ª linha do grafico barHeight = -1 * weatherData[count].temperature; box(xPosition, yPosition, barWidth, barHeight); text(xPosition - 20, yPosition - 20 ,weatherData[count].date); } count++; } } function drawing() { color("red"); //monte o grafico com as posições x,y, barChart(-500, 9, 100); color("blue"); //monte o grafico barChart(-100, 9, 100); } // As seguintes funções estão disponíveis: // // color (string) - define a cor atual // lineWidth (number) - define a largura da linha // box (x, y, width, height) - desenhe uma caixa // circle (x, y, radius) - desenhe um círculo // line (x1, y1, x2, y2) - desenhe uma linha // text (x, y, string) - desenhe texto // clear () - limpa a tela // path (string) - desenhe uma linha complexa // Em uma descrição de linha, os seguintes comandos são válidos: // g x y - vai para o ponto x, y sem desenhar // l x y - desenha uma linha do ponto atual para o ponto x, y // c - desenha uma linha de volta ao início da linha // q x y cx cy - desenha uma curva para x, y, usando cx, cy as // 'ponto de controle' para determinar a curvatura // // fill () - preenche o caminho atual com a cor atual // // Coordenadas são interpretadas como se 0,0 é o centro do // tela. x é o eixo horizontal e y a vertical. // Positivo x vai para a direita, positivo y sobe. // Essas operações podem transformar o sistema de coordenadas: // // moveTo (x, y) - move a origem para x, y // rotate (degrees) - rotaciona as operações de desenho subseqüentes // por vários graus // escala (fator) - escala operações de desenho subseqüentes // goBack () - desfaz uma transformação
